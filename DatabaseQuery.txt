SELECT u.user_id, u.username, td.training_id, DATE(td.training_date) AS training_date, Count(*) AS count
FROM "User" u
JOIN Training_details td ON u.user_id = td.user_id
WHERE (u.user_id, td.training_id, DATE(td.training_date)) IN (
    SELECT user_id, training_id, DATE(training_date)
    FROM Training_details
    GROUP BY user_id, training_id, DATE(training_date)
    HAVING COUNT(*) > 1
)
GROUP BY u.user_id, u.username, td.training_id, DATE(td.training_date)
ORDER BY training_date DESC;
